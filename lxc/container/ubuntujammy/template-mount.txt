# mount folder
lxc.mount.entry = /home/selene/debianthinkpad/lxc/container/ubuntujammy/mount  home/carlos none bind,create=dir 0 0


#mount usb serial
lxc.mount.entry = /dev/ttyUSB0 dev/ttyUSB0 none bind,optional,create=file 0 0

# Allow USB devices
lxc.cgroup2.devices.allow = c 189:* rwm
lxc.cgroup2.devices.allow = c 180:* rwm
lxc.cgroup2.devices.allow = c 188:* rwm

# Bind semua USB bus
lxc.mount.entry = /dev/bus/usb dev/bus/usb none bind,create=dir 0 0

# Tambahkan user ke group plugdev/dialout (sesuai device)
usermod -aG plugdev carlos
usermod -aG dialout carlos
